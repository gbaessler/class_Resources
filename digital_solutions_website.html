Foundation

Objects & Classes

In Object-Oriented Programming (OOP), a class is a blueprint and an object is an instance of that blueprint. This mirrors how we model real-world entities in software.

JavaScript – Class Example

JAVASCRIPT
// Define a class – the blueprint
class Animal {
  constructor(name, sound) {
    this.name  = name;   // instance property
    this.sound = sound;
  }

  speak() {
    return `${this.name} says ${this.sound}!`;
  }
}

// Inheritance – Dog extends Animal
class Dog extends Animal {
  constructor(name) {
    super(name, "Woof");   // call parent constructor
    this.tricks = [];
  }

  learnTrick(trick) {
    this.tricks.push(trick);
  }

  showTricks() {
    return `${this.name} knows: ${this.tricks.join(", ")}`;
  }
}

// Create objects (instances)
const myDog = new Dog("Rex");
myDog.learnTrick("sit");
myDog.learnTrick("roll over");

console.log(myDog.speak());       // Rex says Woof!
console.log(myDog.showTricks());  // Rex knows: sit, roll over

Sorting Algorithm

Bubble Sort

Bubble Sort repeatedly steps through a list, compares adjacent elements, and swaps them if they are in the wrong order. The largest values "bubble up" to the end with each pass.

ALGORITHM BubbleSort(array)
    n ← LENGTH(array)
    FOR i ← 0 TO n - 1 DO
        FOR j ← 0 TO n - i - 2 DO
            IF array[j] > array[j+1] THEN
                SWAP array[j] AND array[j+1]
            END IF
        END FOR
    END FOR
END ALGORITHM

JAVASCRIPT
function bubbleSort(arr) {
  const n = arr.length;

  for (let i = 0; i < n - 1; i++) {
    let swapped = false;

    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        // Destructuring swap
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
        swapped = true;
      }
    }

    // Optimisation: stop if already sorted
    if (!swapped) break;
  }

  return arr;
}

const data = [64, 34, 25, 12, 22, 11, 90];
console.log("Sorted:", bubbleSort(data));
PYTHON
def bubble_sort(arr):
    n = len(arr)

    for i in range(n - 1):
        swapped = False

        for j in range(n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True

        # Optimisation: stop if no swaps made
        if not swapped:
            break

    return arr

data = [64, 34, 25, 12, 22, 11, 90]
print("Sorted:", bubble_sort(data))

Arrays in JavaScript & Python

An array (called a list in Python) is an ordered collection of elements accessed by index. Arrays are zero-indexed — the first element is at position 0.

JAVASCRIPT
// Declare and initialise an array
const scores = [85, 92, 78, 95, 88];

// Access elements by index (zero-based)
console.log(scores[0]);   // 85  (first element)
console.log(scores[4]);   // 88  (last element)

// Array methods
scores.push(100);         // Add to end
scores.pop();             // Remove from end
scores.unshift(70);      // Add to start
scores.shift();           // Remove from start

// Iterating with forEach
scores.forEach((score, index) => {
  console.log(`scores[${index}] = ${score}`);
});

// map, filter, reduce
const doubled = scores.map(s => s * 2);
const passes  = scores.filter(s => s >= 80);
const total   = scores.reduce((sum, s) => sum + s, 0);

// Slicing
const first3 = scores.slice(0, 3);  // [85, 92, 78]

Event Listeners

Event listeners allow your page to respond to user actions — clicks, key presses, mouse moves, and more. They are the backbone of interactive web pages.

JAVASCRIPT – Event Listener Syntax
// Syntax: element.addEventListener(event, handlerFunction)

const button = document.getElementById("myBtn");

button.addEventListener("click", function(event) {
  console.log("Button clicked!", event);
});

// Arrow function shorthand
button.addEventListener("click", (e) => {
  e.target.textContent = "Clicked!";
});

// Keyboard events
document.addEventListener("keydown", (e) => {
  console.log(`Key pressed: ${e.key}`);
});

// Mouse events
box.addEventListener("mouseover", () => box.style.background = "blue");
box.addEventListener("mouseout",  () => box.style.background = "");

// Form events
input.addEventListener("input", (e) => console.log(e.target.value));
form.addEventListener("submit", (e) => { e.preventDefault(); });

Algorithm Design

Euclid's Tiles Problem

Given a rectangle of width W and height H, what is the largest square tile that can perfectly tile the rectangle with no gaps or overlaps? The answer is the GCD (Greatest Common Divisor) of W and H.

📋 Step-by-Step Algorithm

  1. Understand the problem: We need to find the largest square that divides evenly into both the width and height of the rectangle.
  2. Identify the pattern: This is the Euclidean Algorithm — repeatedly subtract the smaller value from the larger until they are equal.
  3. Write the pseudocode: Express the logic in plain language before coding it.
  4. Trace through an example: Use W=12, H=8 → 12-8=4, 8-4=4, 4=4 ✓ → GCD = 4.
  5. Implement in code: Translate pseudocode into JavaScript or Python.
  6. Test with edge cases: Try W=H (same dimensions), W=1, prime numbers.
  7. Visualise the solution: Use canvas/turtle to draw the tiles.
ALGORITHM EuclidTiles(width, height)
    WHILE width ≠ height
        IF width > height THEN
            width ← width − height
        ELSE
            height ← height − width
        END IF
    END WHILE
    RETURN width   // width = height = GCD = tile size
END ALGORITHM

▶ Interactive Euclid's Tiles

JAVASCRIPT
// Subtraction-based (matches algorithm exactly)
function euclidSubtract(w, h) {
  while (w !== h) {
    if (w > h) w -= h;
    else h -= w;
  }
  return w;  // w === h === GCD
}

// Efficient version using modulo
function gcd(a, b) {
  while (b !== 0) {
    [a, b] = [b, a % b];
  }
  return a;
}

const tileSize = gcd(360, 252);
console.log(`Tile size: ${tileSize}`);          // 36
console.log(`Tiles across: ${360 / tileSize}`); // 10
console.log(`Tiles down: ${252 / tileSize}`);   // 7
PYTHON
# Subtraction-based (matches pseudocode directly)
def euclid_subtract(w, h):
    while w != h:
        if w > h:
            w -= h
        else:
            h -= w
    return w

# Efficient version using modulo
def gcd(a, b):
    while b:
        a, b = b, a % b
    return a

# Recursive version
def gcd_recursive(a, b):
    if b == 0:
        return a
    return gcd_recursive(b, a % b)

tile_size = gcd(360, 252)
print(f"Tile size: {tile_size}")           # 36
print(f"Tiles: {360//tile_size} × {252//tile_size}")
